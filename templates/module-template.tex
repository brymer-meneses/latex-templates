\newcommand{\defineCustomHeading}[2]{
    % Usage:
    %       \defineCustomHeading{command_alias}{heading_mark}
    % Example:
    %       \defineCustomHeading{mod}{Module}
    %       \mod{3-4}
    %
    \expandafter\newcommand\csname #1\endcsname[1]{
        \begin{center}
            \noindent
            \LARGE \textbf{#2:} ##1 \\[1cm]
        \end{center}
    }
}

\newcommand{\defineNumberedHeading}[2]{%
    % Usage:
    %       \defineNumberedHeading{command_alias}{heading_mark}
    % Example:
    %       \defineNumberedHeading{lesson}{Lesson}
    %       \lesson{}
    %
    \newcounter{#1num}
    \expandafter\newcommand\csname #1\endcsname[1]{%
        \stepcounter{#1num}
        \begin{center}
            \LARGE \textbf{#2 \csname the#1num\endcsname}: \\[2mm]
            ##1
        \end{center}
    }
}
